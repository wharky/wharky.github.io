### 🍝 문제: '스파게티 내러티브'의 저주

초기 IMGs 개발은 대개 간단한 `if-else` 분기문이나 거대한 `switch` 문으로 시작됩니다. "만약 플레이어가 A를 선택했다면 이 장면으로, B를 선택했다면 저 장면으로…" 이런 식이죠. 처음에는 괜찮습니다. 스토리 라인이 몇 개 안 되니까요. 하지만 스토리가 조금만 복잡해지고, 선택지가 서로 다른 결과에 영향을 미치기 시작하면?

*   **수십 개의 분기**: `if-else`가 끝없이 중첩되고, `goto` 문이 난무하며 코드는 걷잡을 수 없이 엉킵니다. 특정 버그를 찾으려면 전체 서사 흐름을 머릿속으로 시뮬레이션해야 하죠.
*   **유지보수의 악몽**: 디자이너가 "이 대사 다음 선택지를 하나 더 추가해주세요"라고 말하는 순간, 당신은 이미 영혼이 빠져나가는 것을 느낍니다. 새로운 기능을 추가하거나 기존 로직을 변경하는 것은 마치 거미줄에서 한 가닥을 끊으려다 전체를 망가뜨리는 것과 같습니다.
*   **확장성의 한계**: 서사 스케일이 커지거나 새로운 스토리라인을 추가할 때마다 기존 코드를 대폭 수정해야 합니다. 결국 '처음부터 다시 만들자'는 말이 나오기 십상이죠.
*   **테스트의 어려움**: 모든 분기점을 테스트하는 것은 거의 불가능에 가깝습니다. 특정 조건에서만 발생하는 버그는 출시 후에야 발견되어 플레이어의 뒷목을 잡게 만들죠.

이러런 문제들은 단순한 코딩 스타일의 문제가 아닙니다. 이는 시스템의 근본적인 설계 부재에서 오는 구조적 문제입니다. 그리고 우리는 여기에 **Game Design Patterns**라는 강력한 도구를 적용하여 해결책을 찾아낼 수 있습니다.

### ✨ 해결책: 게임 디자인 패턴으로 '모듈화된 이야기' 시스템 구축하기

제가 IMGs 시스템 구축에 가장 강력하게 추천하는 패턴은 바로 **State Pattern**입니다. 인터랙티브 무비 게임의 본질은 결국 '다양한 상태(장면, 대화, 선택지)를 오가는 유한 상태 머신(Finite State Machine, FSM)'으로 볼 수 있기 때문이죠.

#### 🎯 핵심 패턴: State Pattern (상태 패턴)

State Pattern은 객체의 내부 상태가 변경될 때 객체의 동작을 변경하도록 허용합니다. 객체는 마치 자신의 클래스를 변경하는 것처럼 보이죠. IMGs에서는 각 '장면(Scene)' 또는 '서사 단계(Narrative Segment)'를 하나의 상태로 정의할 수 있습니다.

**문제 해결 방식:**

1.  **상태 추상화**: 모든 서사 상태(대화, 선택, 컷신, 액션 등)가 공통으로 가져야 할 인터페이스를 정의합니다.
2.  **구체적인 상태 구현**: 각 서사 단계를 해당 인터페이스를 구현하는 별도의 클래스로 만듭니다.
3.  **상태 머신 (Context)**: 현재 상태를 관리하고, 외부 입력에 따라 다음 상태로 전환하는 로직을 담당하는 핵심 클래스를 만듭니다.

// 1. 모든 서사 상태가 구현해야 할 인터페이스 (예: C#)
public interface INarrativeState
{
    void Enter(); // 상태 진입 시 호출
    void HandleInput(PlayerChoice choice); // 플레이어 입력 처리
    void Update(); // 매 프레임 업데이트 (선택 사항)
    void Exit(); // 상태 종료 시 호출
}

// 2-1. 대화 상태
public class DialogueState : INarrativeState
{
    private NarrativeStateMachine _stateMachine;
    private DialogueSegment _currentDialogue; // 현재 보여줄 대사 데이터

    public DialogueState(NarrativeStateMachine machine, DialogueSegment dialogue)
    {
        _stateMachine = machine;
        _currentDialogue = dialogue;
    }

    public void Enter()
    {
        Console.WriteLine($"[대화] {_currentDialogue.Speaker}: {_currentDialogue.Text}");
        // UI에 대사 표시, 캐릭터 애니메이션 시작 등
    }

    public void HandleInput(PlayerChoice choice)
    {
        if (choice == PlayerChoice.ContinueDialogue) // 다음 대사로 넘어가기 선택
        {
            DialogueSegment nextDialogue = GetNextDialogue(_currentDialogue);
            if (nextDialogue != null)
            {
                _currentDialogue = nextDialogue;
                Enter(); // 다음 대사로 갱신하여 다시 표시
            }
            else
            {
                // 모든 대사가 끝났으면 다음 상태로 전환 (예: 선택지 상태로)
                _stateMachine.ChangeState(new ChoiceState(_stateMachine, GetChoicesForCurrentScene()));
            }
        }
    }

    public void Exit() { Console.WriteLine("대화 종료."); /* UI 정리 등 */ }

// 2-2. 선택지 상태
public class ChoiceState : INarrativeState
{
    private NarrativeStateMachine _stateMachine;
    private List<Choice> _availableChoices; // 플레이어에게 보여줄 선택지 목록

    public ChoiceState(NarrativeStateMachine machine, List<Choice> choices)
    {
        _stateMachine = machine;
        _availableChoices = choices;
    }

    public void Enter()
    {
        Console.WriteLine("\
[선택지]");
        foreach (var choice in _availableChoices)
        {
            Console.WriteLine($"- {choice.Text}");
        }
        // UI에 선택지 표시
    }

    public void HandleInput(PlayerChoice playerChoice)
    {
        // 플레이어의 선택에 따라 다음 상태로 전환
        Choice selected = _availableChoices.FirstOrDefault(c => c.ID == playerChoice.ID);
        if (selected != null)
        {
            Console.WriteLine($"플레이어 선택: {selected.Text}");
            // 선택에 따른 결과 처리 (예: 게임 변수 변경, 관계도 변화 등)
            _stateMachine.GameContext.ProcessChoiceConsequence(selected);

            // 중요: 다음 상태는 선택지에 따라 동적으로 결정됨
            // 이를 위해 팩토리나 전략 패턴을 활용할 수 있음
            _stateMachine.ChangeState(selected.NextStateFactory.CreateState(_stateMachine));
        }
        else
        {
            Console.WriteLine("잘못된 선택입니다.");
        }
    }

    public void Update() { /* 입력 대기 애니메이션 등 */ }
    public void Exit() { Console.WriteLine("선택지 종료."); /* UI 정리 등 */ }
}

// 3. 상태 머신 (Context) - 서사 흐름의 핵심
public class NarrativeStateMachine
{
    public INarrativeState CurrentState { get; private set; }
    public GameContext GameContext { get; private set; } // 게임 전반의 데이터 (플레이어 변수, 관계 등)

    public NarrativeStateMachine(GameContext context)
    {
        GameContext = context;
        // 게임 시작 시 초기 상태 설정
        ChangeState(new DialogueState(this, GameContext.StoryData.GetInitialDialogue()));
    }

    public void ChangeState(INarrativeState newState)
    {
        CurrentState?.Exit(); // 현재 상태 종료 처리
        CurrentState = newState;
        CurrentState.Enter(); // 새 상태 진입 처리
    }

    public void HandlePlayerInput(PlayerChoice choice)
    {
        CurrentState?.HandleInput(choice);
    }

    public void Update()
    {
        CurrentState?.Update();
    }
}
